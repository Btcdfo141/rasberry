This PR contains the new configuration for all devices.

In order to keep track of the changes, this document contains a chapter
for each device type, indicating the new configuration and how to convert
from old configuration.

binary_sensor:

	modbus:
		binary_sensors: []

	--> NO CHANGES.


sensor:
	modbus:
		sensors: []

	new parameters:
		CONF_INPUT_TYPE, optional (CALL_TYPE_COIL) CALL_TYPE_DISCRETE or CALL_TYPE_COIL
		CONF_ADDRESS, required

	update route:
		CONF_REGISTER --> CONF_ADDRESS
		CONF_REGISTER_TYPE --> CONF_INPUT_TYPE

switch:
	modbus:
		switches: []

	new parameters:
		CONF_ADDRESS, required (was CALL_TYPE_COIL and CONF_REGISTER)
		CONF_DEVICE_CLASS, optional
		CONF_INPUT_TYPE, optional (CALL_TYPE_REGISTER_HOLDING)
			CALL_TYPE_REGISTER_HOLDING, CALL_TYPE_REGISTER_INPUT, CALL_TYPE_COIL
		CONF_COMMAND_OFF, optional (0x00)
		CONF_COMMAND_ON, optional (0x01)

		CONF_VERIFY_STATE removed
	update route:
		CALL_TYPE_COIL -> CONF_ADDRESS and CONF_INPUT_TYPE=CONF_COIL
		CONF_REGISTER -> CONF_ADDRESS
		CONF_VERIFY_STATE automatic when CONF_VERIFY_REGISTER is used

		CONF_REGISTER_TYPE -> CONF_INPUT_TYPE
