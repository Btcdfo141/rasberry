This PR contains the new configuration for all devices.

In order to keep track of the changes, this document contains a chapter
for each device type, indicating the new configuration and how to convert
from old configuration.

binary_sensor:

	modbus:
		binary_sensors: []

	--> NO CHANGES.


sensor:
	modbus:
		sensors: []

	new parameters:
		CONF_INPUT_TYPE, optional (CALL_TYPE_COIL) CALL_TYPE_DISCRETE or CALL_TYPE_COIL
		CONF_ADDRESS, required

	update route:
		CONF_REGISTER --> CONF_ADDRESS
		CONF_REGISTER_TYPE --> CONF_INPUT_TYPE
